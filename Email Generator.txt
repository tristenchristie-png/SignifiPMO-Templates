        // Generate UI Email
        function generateUIEmail() {
            const form = document.getElementById('uiForm');
            const formData = new FormData(form);
            const data = {};
            
            // Handle regular form data
            for (let [key, value] of formData.entries()) {
                if (key.includes('workflows') || key.includes('languages') || key.includes('branding')) {
                    // Handle checkboxes separately
                    continue;
                }
                data[key] = value || 'N/A';
            }
            
            // Handle checkboxes
            const workflows = Array.from(form.querySelectorAll('input[name="ui_workflows"]:checked')).map(cb => cb.value);
            const languages = Array.from(form.querySelectorAll('input[name="ui_languages"]:checked')).map(cb => cb.value);
            const branding = Array.from(form.querySelectorAll('input[name="ui_branding"]:checked')).map(cb => cb.value);
            
            // Combine workflows
            let workflowList = workflows.join(', ');
            if (data.ui_workflowOther && data.ui_workflowOther !== 'N/A') {
                workflowList += (workflowList ? ', ' : '') + data.ui_workflowOther;
            }
            
            // Combine languages
            let languageList = languages.join(', ');
            if (data.ui_languageOther && data.ui_languageOther !== 'N/A') {
                languageList += (languageList ? ', ' : '') + data.ui_languageOther;
            }
            
            // Combine branding
            let brandingList = branding.join(', ');
            if (data.ui_brandingOther && data.ui_brandingOther !== 'N/A') {
                brandingList += (brandingList ? ', ' : '') + data.ui_brandingOther;
            }
            
            const emailContent = `Subject: UI Request

UI Request

CP: ${data.ui_cp}
CX: ${data.ui_cx}
Kiosk Type: ${data.ui_kioskType}
Vending Unit Elevator: ${data.ui_vendingElevator}
Screen Size: ${data.ui_screenSize}
Smart Workflows: ${workflowList || 'N/A'}
Authentication: ${data.ui_authentication}
Payment Device: ${data.ui_paymentDevice}
Language: ${languageList || 'English (Default)'}
ADA Keypad: ${data.ui_adaKeypad}
API Integration: ${data.ui_apiIntegration}
Hosting: ${data.ui_hosting}
Vision URL: ${data.ui_visionUrl}
Attachments: ${data.ui_attachments}
Branding Documents: ${brandingList || 'N/A'}
SOW: ${data.ui_sow}`;
            
            displayEmail(emailContent);
        }
        
        // Helper functions
        function formatDate(dateString) {
            if (!dateString) return 'N/A';
            const date = new Date(dateString);
            return date.toLocaleDateString('en-US', { 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric' 
            });
        }
        
        function displayEmail(content) {
            document.getElementById('emailContent').textContent = content;
            document.getElementById('emailPreview').style.display = 'block';
            document.getElementById('emailPreview').scrollIntoView({ behavior: 'smooth' });
        }
        
        // Auto-save functionality
        const allInputs = document.querySelectorAll('input, select, textarea');
        
        allInputs.forEach(input => {
            // Load saved data
            const savedValue = localStorage.getItem(`form_${input.name}`);
            if (savedValue && input.type !== 'date') {
                if (input.type === 'checkbox') {
                    input.checked = savedValue === 'true';
                } else {
                    input.value = savedValue;
                }
            }
            
            // Save data on change
            input.addEventListener('change', () => {
                if (input.type === 'checkbox') {
                    localStorage.setItem(`form_${input.name}`, input.checked);
                } else {
                    localStorage.setItem(`form_${input.name}`, input.value);
                }
            });
        });
